```bash

# (On Debian & Ubuntu linux, you may need to `apt install python-venv` first.)
python3 -m venv venv
source venv/bin/activate
pip install -r requirements.txt

# On Windows:
python -m venv venv
venv\Scripts\activate
pip install -r requirements.txt
```

